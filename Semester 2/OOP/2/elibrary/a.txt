import 'dart:io';

class Student {
  List<String> dataMahasiswa;
  String nama;
  int nim;

  Student(this.nama, this.nim);
}

class LibrarysTransaction {
  Student mahasiswa;
  String judulBuku;
  DateTime date;

  LibrarysTransaction(this.mahasiswa, this.judulBuku, this.date);
}

class Book {
  List<String> dataBuku;

  Book(this.dataBuku);

  List<LibrarysTransaction> dataTransaksi = [];
  // Method
  void pinjamBuku(Student mahasiswa, String judulBuku, DateTime date) {
    if (dataBuku.contains(judulBuku)) {
      dataBuku.remove(judulBuku);
      dataTransaksi.add(LibrarysTransaction(mahasiswa, judulBuku, date));
      print('${mahasiswa.nama} meminjam $judulBuku.');
    } else {
      print('Buku $judulBuku tidak tersedia.');
    }
  }

  void kembalikan(Student mahasiswa, String judulBuku, DateTime date) {
    if (dataTransaksi.any((transaksi) =>
        transaksi.mahasiswa == mahasiswa && transaksi.judulBuku == judulBuku)) {
      dataBuku.add(judulBuku);
      print(
          '${mahasiswa.nama} mengembalikan Buku $judulBuku pada tanggal ${date.toString()}');
      dataTransaksi.removeWhere((transaksi) =>
          transaksi.mahasiswa == mahasiswa && transaksi.judulBuku == judulBuku);
    } else {
      print('${mahasiswa.nama} did not borrow $judulBuku.');
    }
  }

  }
  void tambahBuku() {
    // TODO
  }
}
